# SkyScouter Jetson runtime requirements

# Use this after installing the JetPack-matched NVIDIA PyTorch/torchvision
# wheels or NVIDIA container for your Jetson. Do not use the Windows/CUDA 12.8
# workstation requirements file as the source of truth on Jetson.
#
# Recommended order on Jetson:
#   python3 -m venv .venv_jetson --system-site-packages
#   source .venv_jetson/bin/activate
#   python -m pip install --upgrade pip setuptools wheel
#   python -m pip install -r requirements-jetson.txt

numpy>=1.23,<3

# Use opencv-python (GUI-capable build) not opencv-python-headless.
# Headless cannot open display windows (operator view crashes).
# System python3-opencv from JetPack is compiled against NumPy 1.x and
# fails with NumPy 2.x in this venv — do not rely on it.
opencv-python>=4.7,<5

# Install PyTorch and torchvision from NVIDIA's Jetson channel first.
# Ultralytics should see that Jetson PyTorch is already installed.

# ONNX is required as an intermediate format for TensorRT engine export
# (PyTorch → ONNX → TRT).  onnxruntime-gpu is NOT installed — on Jetson
# TensorRT is the runtime, and onnxruntime-gpu has no aarch64 PyPI wheel.
onnx>=1.12.0,<2.0.0
onnxslim>=0.1.71

# Flight control link (Phase 1+).  The MAVLink interception layer talks to the
# Pixhawk over a serial link via pymavlink, which uses pyserial underneath for
# the serial transport.  Both are pure-python (aarch64 wheels available) and do
# NOT touch the JetPack CUDA stack.  mavlink_link.py imports pymavlink lazily, so
# the perception-only pipeline still runs if these are absent — but the FC
# options (menu 5/6) need them.
pyserial>=3.5,<4
pymavlink>=2.4.40,<3

# Development and regression checks on the Jetson venv.
pytest>=7.4,<9
